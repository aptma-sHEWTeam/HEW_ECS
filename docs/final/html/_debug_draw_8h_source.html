183 void DrawGrid(float size = 10.0f, int divisions = 10, const DirectX::XMFLOAT3& color = {0.5f, 0.5f, 0.5f}) {
184 float step = size / divisions;
185 float halfSize = size * 0.5f;
186
187 // X-Z平面のグリッド
188 for (int i = 0; i <= divisions; ++i) {
189 float pos = -halfSize + i * step;

193 DirectX::XMFLOAT3{-halfSize, 0, pos},

199 AddLine(
200 DirectX::XMFLOAT3{pos, 0, -halfSize},

242 void Render(GfxDevice& gfx, const Camera& cam) {
243 if (lines_.empty()) return;
244
245 // 頂点データを構築
246 std::vector<Vertex> vertices;
247 vertices.reserve(lines_.size() * 2);
248
249 for (const auto& line : lines_) {
250 vertices.push_back({ line.start, line.color });
251 vertices.push_back({ line.end, line.color });
252 }
253
254 // 頂点バッファを更新
255 D3D11_MAPPED_SUBRESOURCE mapped;
256 if (SUCCEEDED(gfx.Ctx()->Map(vb_.Get(), 0, D3D11_MAP_WRITE_DISCARD, 0, &mapped))) {
257 memcpy(mapped.pData, vertices.data(), vertices.size() * sizeof(Vertex));
258 gfx.Ctx()->Unmap(vb_.Get(), 0);
259 }
260
261 // パイプライン設定
262 gfx.Ctx()->IASetInputLayout(layout_.Get());
263 gfx.Ctx()->VSSetShader(vs_.Get(), nullptr, 0);
264 gfx.Ctx()->PSSetShader(ps_.Get(), nullptr, 0);
265 gfx.Ctx()->VSSetConstantBuffers(0, 1, cb_.GetAddressOf());
266
267 UINT stride = sizeof(Vertex);
268 UINT offset = 0;
269 gfx.Ctx()->IASetVertexBuffers(0, 1, vb_.GetAddressOf(), &stride, &offset);
270 gfx.Ctx()->IASetPrimitiveTopology(D3D11_PRIMITIVE_TOPOLOGY_LINELIST);
271
272 // 定数バッファ更新（ワールド行列は単位行列）
273 DirectX::XMMATRIX VP = DirectX::XMMatrixTranspose(cam.View * cam.Proj);
274 gfx.Ctx()->UpdateSubresource(cb_.Get(), 0, nullptr, &VP, 0, 0);
275
276 // 描画
277 gfx.Ctx()->Draw((UINT)vertices.size(), 0);
278 }
