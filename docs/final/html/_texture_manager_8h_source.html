111 TextureHandle LoadFromFile(const char* filepath) {
112 // WICを使用して画像を読み込む
113 Microsoft::WRL::ComPtr<IWICImagingFactory> wicFactory;
114 HRESULT hr = CoCreateInstance(
115 CLSID_WICImagingFactory,
116 nullptr,
117 CLSCTX_INPROC_SERVER,
118 IID_PPV_ARGS(&wicFactory)
119 );
120
121 if (FAILED(hr)) {
122 char msg[512];
123 sprintf_s(msg, "Failed to create WIC factory for: %s", filepath);
124 MessageBoxA(nullptr, msg, "Texture Load Error", MB_OK | MB_ICONERROR);
125 return INVALID_TEXTURE;
126 }
127
128 // ワイド文字列に変換
129 wchar_t wpath[MAX_PATH];
130 MultiByteToWideChar(CP_ACP, 0, filepath, -1, wpath, MAX_PATH);
131
132 // デコーダーを作成
133 Microsoft::WRL::ComPtr<IWICBitmapDecoder> decoder;
134 hr = wicFactory->CreateDecoderFromFilename(
135 wpath,
136 nullptr,
137 GENERIC_READ,
138 WICDecodeMetadataCacheOnDemand,
139 &decoder
140 );
141
142 if (FAILED(hr)) {
143 char msg[512];
144 sprintf_s(msg, "Failed to load image file: %s", filepath);
145 MessageBoxA(nullptr, msg, "Texture Load Error", MB_OK | MB_ICONERROR);
146 return INVALID_TEXTURE;
147 }
148
149 // フレームを取得
150 Microsoft::WRL::ComPtr<IWICBitmapFrameDecode> frame;
151 hr = decoder->GetFrame(0, &frame);
152 if (FAILED(hr)) return INVALID_TEXTURE;
153
154 // RGBA32に変換
155 Microsoft::WRL::ComPtr<IWICFormatConverter> converter;
156 hr = wicFactory->CreateFormatConverter(&converter);
157 if (FAILED(hr)) return INVALID_TEXTURE;
158
159 hr = converter->Initialize(
160 frame.Get(),
161 GUID_WICPixelFormat32bppRGBA,
162 WICBitmapDitherTypeNone,
163 nullptr,
164 0.0,
165 WICBitmapPaletteTypeCustom
166 );
167 if (FAILED(hr)) return INVALID_TEXTURE;
168
169 // サイズを取得
170 UINT width, height;
171 hr = converter->GetSize(&width, &height);
172 if (FAILED(hr)) return INVALID_TEXTURE;
173
174 // ピクセルデータを取得
175 std::vector<uint8_t> pixels(width * height * 4);
176 hr = converter->CopyPixels(
177 nullptr,
178 width * 4,
179 static_cast<UINT>(pixels.size()),
180 pixels.data()
181 );
182 if (FAILED(hr)) return INVALID_TEXTURE;
183
184 return CreateTextureFromMemory(pixels.data(), width, height, 4);
185 }

199 TextureHandle CreateTextureFromMemory(const uint8_t* data, uint32_t width, uint32_t height, uint32_t channels) {
200 D3D11_TEXTURE2D_DESC texDesc{};
201 texDesc.Width = width;
202 texDesc.Height = height;
203 texDesc.MipLevels = 1;
204 texDesc.ArraySize = 1;
205 texDesc.Format = DXGI_FORMAT_R8G8B8A8_UNORM;
206 texDesc.SampleDesc.Count = 1;
207 texDesc.Usage = D3D11_USAGE_DEFAULT;
208 texDesc.BindFlags = D3D11_BIND_SHADER_RESOURCE;
209
210 D3D11_SUBRESOURCE_DATA initData{};
211 initData.pSysMem = data;
212 initData.SysMemPitch = width * channels;
213
214 Microsoft::WRL::ComPtr<ID3D11Texture2D> texture;
215 HRESULT hr = gfx_->Dev()->CreateTexture2D(&texDesc, &initData, &texture);
216 if (FAILED(hr)) {
217 MessageBoxA(nullptr, "Failed to create texture2D", "Texture Error", MB_OK | MB_ICONERROR);
218 return INVALID_TEXTURE;
219 }
220
221 // シェーダーリソースビューを作成
222 D3D11_SHADER_RESOURCE_VIEW_DESC srvDesc{};
223 srvDesc.Format = texDesc.Format;
224 srvDesc.ViewDimension = D3D11_SRV_DIMENSION_TEXTURE2D;
225 srvDesc.Texture2D.MipLevels = 1;
226
227 Microsoft::WRL::ComPtr<ID3D11ShaderResourceView> srv;
228 hr = gfx_->Dev()->CreateShaderResourceView(texture.Get(), &srvDesc, &srv);
229 if (FAILED(hr)) {
230 MessageBoxA(nullptr, "Failed to create SRV", "Texture Error", MB_OK | MB_ICONERROR);
231 return INVALID_TEXTURE;
232 }
233
234 // テクスチャを登録
235 TextureHandle handle = nextHandle_++;
236 TextureData texData;
237 texData.texture = texture;
238 texData.srv = srv;
239 texData.width = width;
240 texData.height = height;
241 textures_[handle] = texData;
242
243 return handle;
244 }
