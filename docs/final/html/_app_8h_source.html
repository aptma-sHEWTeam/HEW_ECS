107 void Run() {
108 MSG msg{};
109 auto previousTime = std::chrono::high_resolution_clock::now();
110
111 while (msg.message != WM_QUIT) {
112 // Windowsメッセージ処理
113 if (ProcessWindowsMessages(msg)) {
114 continue;
115 }
116
117 // 時間の計算
118 float deltaTime = CalculateDeltaTime(previousTime);
119
120 // 入力の更新
121 input_.Update();
122
123 // ESCキーで終了
124 if (input_.GetKeyDown(VK_ESCAPE)) {
125 PostQuitMessage(0);
126 }
127
128 // シーンの更新
129 sceneManager_.Update(world_, input_, deltaTime);
130
131 // 画面の描画
132 RenderFrame();
133
134 // ウィンドウタイトルにスコア表示
135 UpdateWindowTitle();
136 }

158 bool CreateAppWindow(HINSTANCE hInst, int width, int height) {

263 bool ProcessWindowsMessages(MSG& msg) {
264 if (PeekMessage(&msg, nullptr, 0U, 0U, PM_REMOVE)) {
265 TranslateMessage(&msg);
266 DispatchMessage(&msg);
267 return true;
268 }

277 float CalculateDeltaTime(std::chrono::high_resolution_clock::time_point& previousTime) {
278 auto currentTime = std::chrono::high_resolution_clock::now();
279 std::chrono::duration<float> deltaTime = currentTime - previousTime;
280 previousTime = currentTime;
281 return deltaTime.count();
282 }

336 static LRESULT CALLBACK WndProcStatic(HWND hWnd, UINT msg, WPARAM wp, LPARAM lp) {
337 App* app = nullptr;
338
339 if (msg == WM_NCCREATE) {
340 CREATESTRUCT* cs = reinterpret_cast<CREATESTRUCT*>(lp);
341 app = reinterpret_cast<App*>(cs->lpCreateParams);
342 SetWindowLongPtr(hWnd, GWLP_USERDATA, reinterpret_cast<LONG_PTR>(app));
343 } else {
344 app = reinterpret_cast<App*>(GetWindowLongPtr(hWnd, GWLP_USERDATA));
345 }
346
347 if (app) {
348 return app->WndProc(hWnd, msg, wp, lp);
349 }
350
351 return DefWindowProc(hWnd, msg, wp, lp);
352 }
353
358 LRESULT WndProc(HWND hWnd, UINT msg, WPARAM wp, LPARAM lp) {
359 switch (msg) {
360 case WM_DESTROY:
361 PostQuitMessage(0);
362 return 0;
363
364 case WM_MOUSEWHEEL:
365 input_.OnMouseWheel(GET_WHEEL_DELTA_WPARAM(wp));
366 return 0;
367 }
368
369 return DefWindowProc(hWnd, msg, wp, lp);
370 }
