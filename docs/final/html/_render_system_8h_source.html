116 const char* VS = R"(
117 cbuffer CB : register(b0) {
118 float4x4 gWVP;
119 float4 gUVTransform; // xy=offset, zw=scale
120 };
121 struct VSIn {
122 float3 pos : POSITION;
123 float2 tex : TEXCOORD;
124 };
125 struct VSOut {
126 float4 pos : SV_POSITION;
127 float2 tex : TEXCOORD;
128 };
129 VSOut main(VSIn i){
130 VSOut o;
131 o.pos = mul(float4(i.pos,1), gWVP);
132 o.tex = i.tex * gUVTransform.zw + gUVTransform.xy;
133 return o;
134 }
135 )";

137 const char* PS = R"(
138 cbuffer CB : register(b0) {
139 float4 gColor;
140 float gUseTexture;
141 float3 padding;
142 };
143 Texture2D gTexture : register(t0);
144 SamplerState gSampler : register(s0);
145 struct VSOut {
146 float4 pos : SV_POSITION;
147 float2 tex : TEXCOORD;
148 };
149 float4 main(VSOut i) : SV_Target {
150 if (gUseTexture > 0.5) {
151 return gTexture.Sample(gSampler, i.tex) * gColor;
152 }
153 return gColor;
154 }
155 )";

283 void Render(GfxDevice& gfx, World& w, const Camera& cam) {
284 gfx.Ctx()->IASetInputLayout(layout_.Get());
285 gfx.Ctx()->VSSetShader(vs_.Get(), nullptr, 0);
286 gfx.Ctx()->PSSetShader(ps_.Get(), nullptr, 0);
287 gfx.Ctx()->VSSetConstantBuffers(0, 1, cb_.GetAddressOf());
288 gfx.Ctx()->PSSetConstantBuffers(0, 1, psCb_.GetAddressOf());
289 gfx.Ctx()->PSSetSamplers(0, 1, samplerState_.GetAddressOf());
290 gfx.Ctx()->RSSetState(rasterState_.Get());
291
292 UINT stride = sizeof(DirectX::XMFLOAT3) + sizeof(DirectX::XMFLOAT2);
293 UINT offset = 0;
294 gfx.Ctx()->IASetVertexBuffers(0, 1, vb_.GetAddressOf(), &stride, &offset);
295 gfx.Ctx()->IASetIndexBuffer(ib_.Get(), DXGI_FORMAT_R16_UINT, 0);
296 gfx.Ctx()->IASetPrimitiveTopology(D3D11_PRIMITIVE_TOPOLOGY_TRIANGLELIST);
297
298 w.ForEach<MeshRenderer>([&](Entity e, MeshRenderer& mr) {
299 auto* t = w.TryGet<Transform>(e);
300 if (!t) return;
301
302 // ワールド行列
303 DirectX::XMMATRIX S = DirectX::XMMatrixScaling(t->scale.x, t->scale.y, t->scale.z);
304 DirectX::XMMATRIX R = DirectX::XMMatrixRotationRollPitchYaw(
305 DirectX::XMConvertToRadians(t->rotation.x),
306 DirectX::XMConvertToRadians(t->rotation.y),
307 DirectX::XMConvertToRadians(t->rotation.z));
308 DirectX::XMMATRIX T = DirectX::XMMatrixTranslation(t->position.x, t->position.y, t->position.z);
309 DirectX::XMMATRIX W = S * R * T;
310
311 // VS定数バッファ
312 VSConstants vsCbuf;
313 vsCbuf.WVP = DirectX::XMMatrixTranspose(W * cam.View * cam.Proj);
314 vsCbuf.uvTransform = DirectX::XMFLOAT4{ mr.uvOffset.x, mr.uvOffset.y, mr.uvScale.x, mr.uvScale.y };
315 gfx.Ctx()->UpdateSubresource(cb_.Get(), 0, nullptr, &vsCbuf, 0, 0);
316
317 // PS定数バッファ
318 PSConstants psCbuf;
319 psCbuf.color = DirectX::XMFLOAT4{ mr.color.x, mr.color.y, mr.color.z, 1.0f };
320 psCbuf.useTexture = (mr.texture != TextureManager::INVALID_TEXTURE) ? 1.0f : 0.0f;
321 gfx.Ctx()->UpdateSubresource(psCb_.Get(), 0, nullptr, &psCbuf, 0, 0);
322
323 // テクスチャ設定
324 if (mr.texture != TextureManager::INVALID_TEXTURE && texManager_) {
325 ID3D11ShaderResourceView* srv = texManager_->GetSRV(mr.texture);
326 if (srv) {
327 gfx.Ctx()->PSSetShaderResources(0, 1, &srv);
328 }
329 }
330
331 gfx.Ctx()->DrawIndexed(indexCount_, 0, 0);
332 });
333 }
