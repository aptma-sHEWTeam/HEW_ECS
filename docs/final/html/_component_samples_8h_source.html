153struct Bouncer : Behaviour {
154 float speed = 2.0f;
155 float amplitude = 2.0f;
156 float time = 0.0f;
157 float startY = 0.0f;
158
164 void OnStart(World& w, Entity self) override {
165 // 最初に1回だけ呼ばれる
166 auto* t = w.TryGet<Transform>(self);
167 if (t) {
168 startY = t->position.y; // 開始位置を記録
169 }
170 }
171
178 void OnUpdate(World& w, Entity self, float dt) override {
179 // 毎フレーム呼ばれる
180 time += dt * speed;
181
182 auto* t = w.TryGet<Transform>(self);
183 if (t) {
184 // sin波で上下に跳ねる
185 t->position.y = startY + sinf(time) * amplitude;
186 }
187 }

337struct RandomWalk : Behaviour {
338 float speed = 2.0f;
339 float changeInterval = 2.0f;
340 float timer = 0.0f;
341 DirectX::XMFLOAT3 direction{ 1.0f, 0.0f, 0.0f };
342
348 void OnStart(World& w, Entity self) override {
349 // 最初にランダムな方向を選ぶ
350 ChooseRandomDirection();
351 }
352
359 void OnUpdate(World& w, Entity self, float dt) override {
360 timer += dt;
361
362 // 一定時間ごとに方向転換
363 if (timer >= changeInterval) {
364 timer = 0.0f;
365 ChooseRandomDirection();
366 }
367
368 // 移動
369 auto* t = w.TryGet<Transform>(self);
370 if (!t) return;
371
372 t->position.x += direction.x * speed * dt;
373 t->position.y += direction.y * speed * dt;
374 t->position.z += direction.z * speed * dt;
375
376 // 範囲外に出たら戻す
377 ClampPosition(t);
378 }
379
380private:
384 void ChooseRandomDirection() {
385 // -1.0 ～ 1.0 のランダムな値
386 direction.x = (static_cast<float>(rand()) / RAND_MAX) * 2.0f - 1.0f;
387 direction.y = (static_cast<float>(rand()) / RAND_MAX) * 2.0f - 1.0f;
388 direction.z = (static_cast<float>(rand()) / RAND_MAX) * 2.0f - 1.0f;
389
390 // 正規化（長さを1に）
391 float length = sqrtf(direction.x * direction.x +
392 direction.y * direction.y +
393 direction.z * direction.z);
394 if (length > 0.0f) {
395 direction.x /= length;
396 direction.y /= length;
397 direction.z /= length;
398 }
399 }
400
405 void ClampPosition(Transform* t) {
406 const float range = 10.0f;
407 if (t->position.x < -range) t->position.x = -range;
408 if (t->position.x > range) t->position.x = range;
409 if (t->position.y < -range) t->position.y = -range;
410 if (t->position.y > range) t->position.y = range;
411 if (t->position.z < -range) t->position.z = -range;
412 if (t->position.z > range) t->position.z = range;
413 }
414};
