179inline Entity CreateCubeOldStyle(World& world) {
180 // ステップ1: エンティティを作成
181 Entity cube = world.CreateEntity();
182
183 // ステップ2: Transformコンポーネントを追加
184 Transform transform;
185 transform.position = DirectX::XMFLOAT3{0.0f, -2.0f, 0.0f};
186 transform.rotation = DirectX::XMFLOAT3{0.0f, 0.0f, 0.0f};
187 transform.scale = DirectX::XMFLOAT3{0.5f, 0.5f, 0.5f};
188 world.Add<Transform>(cube, transform);
189
190 // ステップ3: MeshRendererコンポーネントを追加
191 MeshRenderer renderer;
192 renderer.color = DirectX::XMFLOAT3{0.0f, 1.0f, 1.0f}; // シアン
193 world.Add<MeshRenderer>(cube, renderer);
194
195 // ステップ4: Rotatorコンポーネントを追加
196 Rotator rotator;
197 rotator.speedDegY = 90.0f;
198 world.Add<Rotator>(cube, rotator);
199
200 return cube;
201}

222inline void ModifyEntityExample(World& world, Entity entity) {
223 // Transformを取得して変更
224 auto* transform = world.TryGet<Transform>(entity);
225 if (transform) {
226 transform->position.y += 1.0f; // Y座標を1上げる
227 transform->scale = DirectX::XMFLOAT3{2.0f, 2.0f, 2.0f}; // 2倍の大きさに
228 }
229
230 // MeshRendererを取得して色を変更
231 auto* renderer = world.TryGet<MeshRenderer>(entity);
232 if (renderer) {
233 renderer->color = DirectX::XMFLOAT3{1.0f, 1.0f, 1.0f}; // 白に変更
234 }
235
236 // Rotatorを取得して速度を変更
237 auto* rotator = world.TryGet<Rotator>(entity);
238 if (rotator) {

240 }

304inline void CreateGridOfCubes(World& world, int rows = 3, int cols = 3) {
305 const float spacing = 2.5f; // キューブ間の距離
306
307 for (int row = 0; row < rows; ++row) {
308 for (int col = 0; col < cols; ++col) {
309 // 位置を計算（中心を原点に）
310 float x = (col - cols / 2.0f) * spacing;
311 float z = (row - rows / 2.0f) * spacing;
312
313 // 色を計算（位置によって変わる）
314 float r = static_cast<float>(col) / static_cast<float>(cols - 1);
315 float b = static_cast<float>(row) / static_cast<float>(rows - 1);
316
317 // キューブを作成
318 world.Create()
319 .With<Transform>(
320 DirectX::XMFLOAT3{x, 0.0f, z},
321 DirectX::XMFLOAT3{0.0f, 0.0f, 0.0f},
322 DirectX::XMFLOAT3{0.8f, 0.8f, 0.8f}
323 )
324 .With<MeshRenderer>(DirectX::XMFLOAT3{r, 0.5f, b})
325 .With<Rotator>(45.0f + static_cast<float>(row * 10 + col * 5))
326 .Build();
327 }
328 }
