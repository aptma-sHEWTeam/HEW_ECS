162public:
170 void OnEnter(World& world) override {
171 // 乱数シードを設定
172 srand(static_cast<unsigned int>(time(nullptr)));
173
174 // プレイヤーを作成
175 playerEntity_ = world.Create()
176 .With<Transform>(
177 DirectX::XMFLOAT3{0.0f, -6.0f, 0.0f}, // 画面下部
178 DirectX::XMFLOAT3{0.0f, 0.0f, 0.0f},
179 DirectX::XMFLOAT3{1.0f, 1.0f, 1.0f}
180 )
181 .With<MeshRenderer>(DirectX::XMFLOAT3{0.0f, 1.0f, 0.0f}) // 緑色
182 .With<Player>()
183 .With<PlayerMovement>()
184 .Build();
185
186 score_ = 0;
187 enemySpawnTimer_ = 0.0f;
188 shootCooldown_ = 0.0f;
189 }

198 // プレイヤー移動
199 UpdatePlayerMovement(world, input, deltaTime);
200

202 UpdateShooting(world, input, deltaTime);

221 void OnExit(World& world) override {
222 // 全エンティティを削除(イテレータ破壊を回避)
223 std::vector<Entity> entitiesToDestroy;
224
225 world.ForEach<Transform>([&](Entity e, Transform& t) {
226 entitiesToDestroy.push_back(e);
227 });
228
229 for (const auto& entity : entitiesToDestroy) {
230 world.DestroyEntity(entity);
231 }
232 }

247 void UpdatePlayerMovement(World& world, InputSystem& input, float deltaTime) {
248 auto* playerTransform = world.TryGet<Transform>(playerEntity_);
249 if (!playerTransform) return;
250
251 const float moveSpeed = 8.0f;
252
253 // Aキーで左移動
254 if (input.GetKey('A')) {
255 playerTransform->position.x -= moveSpeed * deltaTime;
256 }
257
258 // Dキーで右移動
259 if (input.GetKey('D')) {
260 playerTransform->position.x += moveSpeed * deltaTime;
261 }
262 }

270 void UpdateShooting(World& world, InputSystem& input, float deltaTime) {
271 shootCooldown_ -= deltaTime;
272
273 // スペースキーで弾を発射(クールダウン中は発射できない)
274 if (input.GetKey(VK_SPACE) && shootCooldown_ <= 0.0f) {
275 auto* playerTransform = world.TryGet<Transform>(playerEntity_);
276 if (playerTransform) {
277 // プレイヤーの位置から弾を発射
278 world.Create()
279 .With<Transform>(
280 DirectX::XMFLOAT3{playerTransform->position.x, playerTransform->position.y + 1.0f, 0.0f},
281 DirectX::XMFLOAT3{0.0f, 0.0f, 0.0f},
282 DirectX::XMFLOAT3{0.3f, 0.5f, 0.3f} // 小さめ
283 )
284 .With<MeshRenderer>(DirectX::XMFLOAT3{1.0f, 1.0f, 0.0f}) // 黄色
285 .With<Bullet>()
286 .With<BulletMovement>()
287 .Build();
288
289 shootCooldown_ = 0.2f; // 0.2秒のクールダウン
290 }
291 }

299 void UpdateEnemySpawning(World& world, float deltaTime) {
300 enemySpawnTimer_ += deltaTime;
301
302 // 1秒ごとに敵を生成
303 if (enemySpawnTimer_ >= 1.0f) {
304 enemySpawnTimer_ = 0.0f;
305

308

329 void CheckCollisions(World& world) {
330 // 削除する予定のエンティティリスト(イテレータ破壊を回避)
331 std::vector<Entity> entitiesToDestroy;
332
333 // 弾と敵の衝突をチェック
334 world.ForEach<Bullet>([&](Entity bulletEntity, Bullet& bullet) {
335 auto* bulletTransform = world.TryGet<Transform>(bulletEntity);
336 if (!bulletTransform) return;
337
338 // この弾が既に削除予定なら処理をスキップ
339 for (const auto& e : entitiesToDestroy) {
340 if (e.id == bulletEntity.id) return;
341 }
342
343 world.ForEach<Enemy>([&](Entity enemyEntity, Enemy& enemy) {
344 auto* enemyTransform = world.TryGet<Transform>(enemyEntity);
345 if (!enemyTransform) return;
346
347 // この敵が既に削除予定なら処理をスキップ
348 for (const auto& e : entitiesToDestroy) {
349 if (e.id == enemyEntity.id) return;
350 }
351
352 // 簡易的な距離判定(円の衝突)
353 float dx = bulletTransform->position.x - enemyTransform->position.x;
354 float dy = bulletTransform->position.y - enemyTransform->position.y;
355 float distance = sqrtf(dx * dx + dy * dy);
356
357 // 衝突したら削除リストに追加してスコア加算
358 if (distance < 1.0f) {
359 entitiesToDestroy.push_back(bulletEntity);
360 entitiesToDestroy.push_back(enemyEntity);
361 score_ += 10;
362 }
363 });
364 });
365
366 // イテレーション後にまとめて削除
367 for (const auto& entity : entitiesToDestroy) {
368 world.DestroyEntity(entity);
369 }
370 }

375 float shootCooldown_;
376};
